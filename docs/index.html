Modeling Bounded Confidence Updating in Julia I

Igor Douven & Rainer Hegselmann

Thanks to the new Agents.jl package for Julia (see here), we can now very easily model bounded confidence updating. The package also offers a lot of functionality out of the box to facilitate running simulations. This is meant as a quick introduction to using that package for replicating most of the results presented in our paper "Mis- and disinformation in a bounded confidence model". Where performance really mattered, we used our own code, which does not rely on the Agents.jl package. (The authors of the Agents.jl package were primarily concerned with generality – the package should [ideally] be usable for any agent-based modeling – and only then with performance.) Our code is available, too.

The simple model

We start by looking at the simple bounded-confidence model and then add truth attraction to it. Refinements and setting up simulations are left to later parts.

Loading the required packages:

using Agents
using Statistics: mean
using Plots

Define the type of agent we need. For more complicated versions of the model, we will want to agent to have more properties, but that is for later:

mutable struct HKAgent <: AbstractAgent
    id::Int
    old_opinion::Float64
    new_opinion::Float64
    previous_opinion::Float64
end

Executing this function will set up the model:

function hk_model(numagents::Int, ϵ::Float64)
    model = ABM(HKAgent, scheduler=fastest, properties=Dict(:ϵ=>ϵ))
    for i in 1:numagents
        o = rand()
        add_agent!(model, o, o, -1)
    end
    return model
end

Now follow a number of helper function, the first defining who is in an agent's BCI, the others defining whow the update is supposed to proceed:

function boundfilter(agent, model)
    filter(j->isapprox(agent.old_opinion, j; rtol=model.ϵ), [ a.old_opinion for a in allagents(model) ])
end

function agent_step!(agent, model)
    agent.previous_opinion = agent.old_opinion
    agent.new_opinion = mean(boundfilter(agent, model))
end

function model_step!(model)
    for a in allagents(model)
        a.old_opinion = a.new_opinion
    end
end

The following is again a helper function, to determine when convergence has been reached:

function terminate(model, s)
    if any(!isapprox(a.previous_opinion, a.new_opinion; rtol=1e-6) for a in allagents(model))
        return false
    else
        return true
    end
end

We associate two methods with model_run: one that runs the model till convergence, and one that allows us to set the number of updates we want.

function model_run(numagents::Int, ϵ::Float64)
    model = hk_model(numagents, ϵ)
    agent_properties = [:new_opinion]
    agent_data, _ = run!(model, agent_step!, model_step!, terminate, adata = agent_properties)
    return agent_data
end

function model_run(numagents::Int, ϵ::Float64, iterations::Int)
    model = hk_model(numagents, ϵ)
    agent_properties = [:new_opinion]
    agent_data, _ = run!(model, agent_step!, model_step!, iterations + 1, adata = agent_properties)
    return agent_data
end

That is the whole set up.

We first consider the case in which we run the model till convergence:

data1 = model_run(50, .2)
plot(data1.step, data1.new_opinion, leg = false, group = data1.id)

Run a pre-determined number of steps:

data2 = model_run(50, .2, 15)
plot(data2.step, data2.new_opinion, leg = false, group = data2.id)

Adding truth

This function sets up the model with truth:

function hk_model(numagents::Int, τ::Float64, ϵ::Float64, α::Float64)
    model = ABM(HKAgent, scheduler=fastest, properties=Dict(:τ=>τ, :ϵ=>ϵ, :α=>α))
    for i in 1:numagents
        o = rand()
        add_agent!(model, o, o, -1)
    end
    return model
end

The helper functions remain the same, except that we need to redefine how an agent's new opinion is calculated, given that they now receive evidence directly about $\tau$:

function agent_step!(agent, model)
    agent.previous_opinion = agent.old_opinion
    agent.new_opinion = model.α*model.τ + (1 - model.α)*mean(boundfilter(agent, model))
end

And again two methods associated with model_run:

function model_run(numagents::Int, τ::Float64, ϵ::Float64, α::Float64)
    model = hk_model(numagents, τ, ϵ, α)
    agent_properties = [:new_opinion]
    agent_data, _ = run!(model, agent_step!, model_step!, terminate, adata = agent_properties)
    return agent_data
end

function model_run(numagents::Int, τ::Float64, ϵ::Float64, α::Float64, iterations::Int)
    model = hk_model(numagents, τ, ϵ, α)
    agent_properties = [:new_opinion]
    when = 0:iterations + 1
    agent_data, _ = run!(model, agent_step!, model_step!, iterations + 1, adata = agent_properties)
    return agent_data
end

Choosing the parameters:

numagents = 50
τ = .7
ϵ = .2
α = .2
numiterations = 15

The case in which we run the model till convergence:

plot(data3.step, data3.new_opinion, leg = false, group = data3.id)

The case with a pre-determined number of steps:
